Let's Learn JavaScript's in built String Functions

Open Console to see the output for the JavaScript code.

The code in the external JS file is as follows.

      
        console.clear();

        var num1 = 15;
        var num2 = 30;

        // =
        num1 = 20;

        // +=
        num1 += 10; // num1 = num1 + 10
        num1 += num2; // num1 = num1 + num2
        console.log(num1);

        // -=
        num1 -= 10; //num1 = num1 - 10
        num1 -= num2; //num1 = num1 - num2
        console.log(num1);

        // *=
        num1 *= 20; //num1 = num1 * 20
        console.log(num1);

        // /=
        num1 /= 10; //num1 = num1 / 10
        console.log(num1);

        // %=
        num1 %= 10; //num1 = num1 % 10
        console.log(num1);
