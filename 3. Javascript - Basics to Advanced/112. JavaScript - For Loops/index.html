        let str3 = "The Quick Brown Fox.";
        let finalStr3 = "";

          let charCode = str3.charCodeAt(i);
          if (charCode >= 65 && charCode <= 90) {

        //Approach 2

        function swapCase(str) {
          let result = '';
          for (let i = 0; i < str.length; i++) {
            let char = str[i];
            if (char >= 'a' && char <= 'z') {
              result += char.toUpperCase(); // Convert lowercase to uppercase
            } else if (char >= 'A' && char <= 'Z') {
              result += char.toLowerCase(); // Convert uppercase to lowercase
            } else {
              result += char; // Keep non-alphabetic characters unchanged
            }
          }
          return result;
        }

        let inputString = "The Quick Brown Fox.";
        let swappedString = swapCase(inputString);

        console.log(swappedString);
